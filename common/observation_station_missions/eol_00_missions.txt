# Technological Advancement - Stone Age
stone_advancement_0 = {
	icon = "GFX_fleet_task_technological_advance_action"
	small_icon = 4

	resources = {
		category = station_observer_missions
		upkeep = {
			society_research = 5
			energy = 5
			food = 5
		}
	}

	monthly_progress = 0.05
	completion_event = advprim.1

	potential = {
		FROM = {
			exists = owner
			owner = { has_country_flag = stone_age }
		}
		NOR = {
			has_valid_civic = civic_machine_terminator
			has_valid_civic = civic_fanatic_purifiers
			has_valid_civic = civic_hive_devouring_swarm
		}
	}

	valid = {
		has_policy_flag = enlightenment_allowed
		custom_tooltip = {
			fail_text = CANNOT_ADVANCE_FANATIC_XENOPHOBE

			FROM = {
				exists = owner

			}
		}
		NOR = {
			has_valid_civic = civic_fanatic_purifiers
			has_valid_civic = civic_hive_devouring_swarm
			has_valid_civic = civic_machine_terminator
		}
