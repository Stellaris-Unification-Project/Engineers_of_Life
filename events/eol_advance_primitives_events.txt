namespace = advprim

### advance primitive civilizations to their next stage

## Stone Age ##

country_event = {
	id = advprim.1
	title = "advprim.1.name" 
	desc = "advprim.1.desc" 
	picture = GFX_evt_alien_cavemen
	show_sound = event_laboratory_sound
	location = FROMFROM
	
	is_triggered_only = yes
	
	immediate = {
        FROMFROM = {
            owner = { save_event_target_as = primitive_country }
            event_target:primitive_country = {
				switch = {
					trigger = has_country_flag
					stone_age = {
						remove_country_flag = stone_age
						root = { set_country_flag = bronze_age }
					}					
                }
                set_primitive_age_effect = { primitive_age = bronze_age }
                change_country_flag = {
                    icon = {
                        category = "special"
                        file = "primitive.dds"
                    }
                    background = {
				        category = "backgrounds"
				        file = "new_dawn.dds"
			         }
			         colors = {
				        "green"
				        "dark_green"
				        "null"
				        "null"
			         }
                }
            }
            observation_outpost = { add_mission_progress = -1.0 }
        }
    }
}
